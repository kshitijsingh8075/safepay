    function parseUPI(text) {
      if (!text.startsWith("upi://pay")) return null;

      const url = new URL(text);
      const params = new URLSearchParams(url.search);

      return {
        upi_id: params.get("pa"),
        name: params.get("pn"),
        amount: params.get("am"),
        currency: params.get("cu"),
      };
    }

    function showResult(decodedText) {
      const data = parseUPI(decodedText);
      const resultDiv = document.getElementById("result");

      if (data) {
        resultDiv.innerHTML = `
          âœ… <strong>Valid UPI QR Code Detected</strong><br>
          ğŸ§¾ Name: ${data.name || 'N/A'} <br>
          ğŸ’¸ UPI ID: ${data.upi_id} <br>

        `;
      } else {

      }
    }

    const html5QrCode = new Html5Qrcode("reader");

    Html5Qrcode.getCameras().then(cameras => {
      if (cameras && cameras.length) {
        html5QrCode.start(
          cameras[0].id,
          {
            fps: 10,
            qrbox: 250
          },
          decodedText => {
            html5QrCode.stop(); // stop after one scan
            showResult(decodedText);
          },
          errorMessage => {
            // ignore scan errors
          }
        );
      }
    }).catch(err => {
